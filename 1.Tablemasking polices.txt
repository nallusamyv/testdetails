GRANT USAGE ON DATABASE DB_SOURCE TO ROLE VIEW_CREATOR_ROLE;
GRANT USAGE ON SCHEMA DB_SOURCE.SOURCE_SCHEMA TO ROLE VIEW_CREATOR_ROLE;
GRANT SELECT ON TABLE DB_SOURCE.SOURCE_SCHEMA.SOURCE_TABLE TO ROLE VIEW_CREATOR_ROLE;

-- Create view in DB_TARGET
USE DATABASE DB_TARGET;
USE SCHEMA TARGET_SCHEMA;

CREATE VIEW VW_SOURCE_TABLE AS
  SELECT *
  FROM DB_SOURCE.SOURCE_SCHEMA.SOURCE_TABLE;

-- Grant access to END_USER_ROLE
GRANT USAGE ON DATABASE DB_TARGET TO ROLE END_USER_ROLE;
GRANT USAGE ON SCHEMA DB_TARGET.TARGET_SCHEMA TO ROLE END_USER_ROLE;
GRANT SELECT ON VIEW DB_TARGET.TARGET_SCHEMA.VW_SOURCE_TABLE TO ROLE END_USER_ROLE;
